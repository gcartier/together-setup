To build setup for Together

#1 copy / replace together-prod from install to setup
   delete setup/together-prod/.jas
   in setup/together-prod
     jas init -open
     jas add .
     cd ..

#2 right-click Together.nsi and choose Compile NSIS Script

#3 test it locally

#4 copy Together Setup.exe to website/limited
   cd to website/limited
   jas sync
   jas upload
   go to montreal
   cd /var/wwww/together/limited
   sudo jas download
